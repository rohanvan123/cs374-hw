```
MinCostInL(w):
    n <- len(w)
    DP <- [MAX_INT] * (len(w) + 1)
    DP[n + 1] <- 0
    for i <- n to 1
        for j <- i + 1 to n
            substr = w[i, j]
            if isStringInL(substr)
                DP[i] = min(DP[i], cost(substr) + DP[j])

    if DP[0] = INT_MAX
        return "w not in L"
    else
        return DP[0]

```
